⚙️ Google Apps Script 設定步驟

1

建立 Google 試算表

前往 Google Sheets 建立新試算表，複製網址列中的 ID（/d/ 後面那一段）。

2

建立 Apps Script

在試算表中點選「擴充功能 → Apps Script」，刪除原有程式碼，貼上以下程式碼，並把第一行的 YOUR_SPREADSHEET_ID 換成你的試算表 ID。

const SPREADSHEET_ID = 'YOUR_SPREADSHEET_ID'; const ORDER_SHEET = '訂單記錄'; const ACCOUNT_SHEET = '帳號管理'; const PRODUCT_SHEET = '商品管理'; function jsonOut(obj){ return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SPREADSHEET_ID); // ── 商品管理 CRUD ── if (data.action === 'addProduct') { const s = ss.getSheetByName(PRODUCT_SHEET); s.appendRow([data.category||'', data.name, data.unit, Number(data.price)||0]); return jsonOut({success:true}); } if (data.action === 'updateProduct') { const s = ss.getSheetByName(PRODUCT_SHEET); const r = parseInt(data.row); s.getRange(r,1).setValue(data.category||''); s.getRange(r,2).setValue(data.name); s.getRange(r,3).setValue(data.unit); s.getRange(r,4).setValue(Number(data.price)||0); return jsonOut({success:true}); } if (data.action === 'deleteProduct') { ss.getSheetByName(PRODUCT_SHEET).deleteRow(parseInt(data.row)); return jsonOut({success:true}); } // ── 訂單寫入 ── let sheet = ss.getSheetByName(ORDER_SHEET); if (!sheet) { sheet = ss.insertSheet(ORDER_SHEET); sheet.appendRow(['訂單編號','下單時間','門市名稱','聯絡人','送貨日期','品項數','品項明細','備註','狀態','最後更新']); sheet.setFrozenRows(1); sheet.getRange(1,1,1,10).setFontWeight('bold').setBackground('#2e7d32').setFontColor('#ffffff'); } const orderId = 'ORD-' + new Date().getTime(); const itemDetail = data.items.map(i => i.name+' x '+i.qty+i.unit).join(' | '); const now = new Date(); const fmt = d => Utilities.formatDate(d,'Asia/Taipei','yyyy/MM/dd HH:mm:ss'); sheet.appendRow([orderId,fmt(now),data.store,data.contact||'', data.deliverDate||'',data.items.length+'項',itemDetail,data.note||'','待確認',fmt(now)]); sheet.autoResizeColumns(1,10); return jsonOut({success:true, orderId}); } catch(err) { return jsonOut({success:false, error:err.toString()}); } } function doGet(e) { const action = e.parameter.action || 'list'; const ss = SpreadsheetApp.openById(SPREADSHEET_ID); // ── 加盟主登入 ── if (action === 'login') { const s = ss.getSheetByName(ACCOUNT_SHEET); if (!s) return jsonOut({success:false}); const vals = s.getDataRange().getValues(); for (let i=1;i({ id:'P'+(i+2), rowIndex:i+2, category:r[0]||'', name:r[1]||'', unit:r[2]||'', price:r[3]||0 })).filter(p=>p.name); return jsonOut({products}); } // ── 取得商品列表（管理員 CRUD 用）── if (action === 'getProductList') { const s = ss.getSheetByName(PRODUCT_SHEET); if (!s||s.getLastRow()<=1) return jsonOut({products:[]}); const vals = s.getDataRange().getValues(); const products = vals.slice(1).map((r,i)=>({ rowIndex:i+2, category:r[0]||'', name:r[1]||'', unit:r[2]||'', price:r[3]||0 })).filter(p=>p.name); return jsonOut({products}); } // ── 更新訂單狀態 ── const sheet = ss.getSheetByName(ORDER_SHEET); if (action === 'updateStatus') { const row = parseInt(e.parameter.row); sheet.getRange(row,9).setValue(e.parameter.status); sheet.getRange(row,10).setValue( Utilities.formatDate(new Date(),'Asia/Taipei','yyyy/MM/dd HH:mm:ss')); return jsonOut({success:true}); } // ── 取得所有訂單 ── if (!sheet||sheet.getLastRow()<=1) return jsonOut({orders:[]}); const vals = sheet.getDataRange().getValues(); const orders = vals.slice(1).map((r,i)=>({ rowIndex:i+2, orderId:r[0], time:r[1].toString(), store:r[2], contact:r[3], deliverDate:r[4], itemCount:r[5], items:r[6], note:r[7], status:r[8], lastUpdate:r[9].toString() })).reverse(); return jsonOut({orders}); }
3

部署為網路應用程式

點選「部署 → 新部署」→ 類型選「網頁應用程式」→ 執行身分選「我」→ 存取權選「所有人」→ 按「部署」。複製產生的網址。

4

貼上網址

將網址貼在下方，儲存後即完成設定。同樣的網址也要貼到「叫貨系統.html」裡的 SCRIPT_URL。
